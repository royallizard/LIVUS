function ajaxGet(url, fn, failFn)
  request = getRequest();
  request.open("GET", url);
  request.onreadystatechange = function () {
      if (request.readyState == 4)
      {
        if (request.status == 200)
        {
          if (fn != null)
            fn(request);
        }
        else if (failFn != null)
          failFn(request);
      }
    }
  request.send(null);
